# ForzaTek AI v2 — Requirements
# ==============================
# Covers Module 0 (Core) + Module 1 (System) + Module 2 (Ingest).
# Later modules will append to this file as they're built — do NOT delete
# anything from here when you add new lines, only add.
#
# Install with:
#     pip install -r requirements.txt
#
# Tested on Python 3.10 and 3.12 on Windows.

# ─── Module 0 (Core) ───
# stdlib only (sqlite3, pathlib, threading, json) — nothing to pin.

# ─── Module 1 (System) ───
# Eel wraps Chromium and provides the JS↔Python bridge.
eel==0.18.1

# FastAPI side server on port 8001 — for streams and routes that Eel can't
# do well (MJPEG, SSE in later modules).
fastapi==0.115.5
uvicorn==0.32.1

# Pydantic is FastAPI's dep already, pinned here so it doesn't drift.
pydantic==2.10.3

# Global hotkey listener (F7/F8/F9).
keyboard==0.13.5

# ─── Module 2 (Ingest) ───
# Numerical arrays for frame buffers, pHash math, dedup ring.
numpy>=1.26,<2.3

# OpenCV: video walking, JPEG encode/decode, pHash DCT, menu-detector
# (Canny + HSV histogram). The "headless" build avoids pulling Qt on
# servers; on dev machines that need a window, swap to opencv-python.
opencv-python-headless>=4.9,<5.0

# Live screen capture on Windows (preferred path in recorder.py).
# Pure-Python fallback is mss; both are listed so the recorder picks
# whichever is available at runtime.
dxcam==0.0.5 ; sys_platform == "win32"
mss>=9.0,<11.0

# Cross-process HTTP for telemetry tag fetching. The recorder polls
# Module 7's /api/telemetry/live but does NOT import telemetry code —
# this is the keep-modules-independent rule in action.
requests>=2.31,<3

# YouTube downloader. Used by video_ingester.py via subprocess.
# Update this regularly — yt-dlp ships breaking site-extractor changes
# every few weeks and an old pin will silently fail to download.
yt-dlp>=2025.1.15

# Eel wraps Chromium and provides the JS↔Python bridge.

# FastAPI side server on port 8001 — for streams and routes that Eel can't

fastapi==0.115.5
uvicorn==0.32.1

# Pydantic is FastAPI's dep already, pinned here so it doesn't drift.
pydantic==2.10.3

# Global hotkey listener (F7/F8/F9). Optional — main.py treats it as
# non-fatal if unavailable. On Linux it needs root, on Windows it just
# works. Keep it in requirements for the Windows target.
keyboard==0.13.5

# ─── Test-time only (pulled in by FastAPI's TestClient) ───
# Listed here so `pip install -r requirements.txt` is sufficient for
# running the test suite, not just for running the app.
httpx==0.28.1
